众数问题

#include <iostream>

using namespace std;
/*
    算法思路：
    1、我们首先假设中间的元素是众数
    2、然后由两边向中间遍历，直到左右都出现值等于众数的数，记录下众数和重数
    3、这样就将一个数组分为三部分，我们再对左右部分执行上述步骤
    4、注意：使用分治策略解决众数问题需要原集合有序
*/
//先进行快速排序
void Quick_Sort(int *arr,int Begin,int End)
{
    if(Begin>End)
    {
        return;
    }
    int tmp=arr[Begin];//基准
    int i=Begin;//定义两个移动指针
    int j=End;
    while(i!=j)
    {
        while(arr[j]>=tmp&&j>i)
        {
            j--;
        }
        while(arr[i]<=tmp&&i<j)
        {
            i++;
        }
        if(j>i)//避免i和j的位置已经到达了同一位置
        {
            //交换i和j位置的元素
            int t=arr[i];
            arr[i]=arr[j];
            arr[j]=t;
        }
        arr[Begin]=arr[i];//交换基准元素和两个指针最后指向的元素
        arr[i]=tmp;
        //继续排序左右两个数组
        Quick_Sort(arr,Begin,i-1);
        Quick_Sort(arr,i+1,End);
    }
}
//分治
int ans=0;//众数的重数
int idx=0;//众数的下标
void GetMajority(int *arr,int l,int r)
{
    if(l>r)
    {
        return;
    }
    int ll=l;//记录原来l的位置
    int rr=r;//记录原来r的位置
    int mid=(l+r)>>1;//>>符号右移一位相当于除2，只允许整数
    while(l<mid&&arr[l]!=arr[mid])l++;//寻找众数的最左边
    while(r>mid&&arr[r]!=arr[mid])r--;//寻找众数的最右边
    if(ans<=r-l+1)
    {
        if(ans==r-l+1)
        {
            idx=min(mid,idx);//选择下标小的
        }
        else
        {
            idx=mid;
        }
        ans=r-l+1;
    }
    if(l-ll>=ans)
    {
        GetMajority(arr,ll,l-1);//对左部分进行分治
    }
    if(rr-r>=ans)
    {
        GetMajority(arr,r+1,rr);//对右部分进行分治
    }


}
int main()
{
    int arr[10];
    cout<<"请输入十个数: ";
    for(int i=0;i<10;i++)
    {
        cin>>arr[i];
    }
    Quick_Sort(arr,0,9);
    GetMajority(arr,0,9);
    cout<<"众数是: "<<arr[idx]<<endl;
    cout<<"重数是: "<<ans<<endl;
    return 0;
}

汉诺塔

#include <iostream>

using namespace std;
/*
    算法思想:
    1、首先把上边的n-1个盘子从A柱借助C柱移动到B柱，
    2、然后把最下面的一个盘子从A柱移到C柱，
    3、最后把B柱的n-1个盘子在借助A柱移到C柱
    4、剩下的n-1个盘子重复以上步骤
*/
void Hanoi(int n,char a,char b,char c)
{
    if(n==1)
    {
        cout<<"将第"<<n<<"个盘子从"<<a<<"移动到"<<c<<endl;
    }
    else
    {
        Hanoi(n-1,a,c,b);//把n-1个圆盘从A柱借助C柱移动到B柱
        cout<<"将第"<<n<<"个盘子从"<<a<<"移动到"<<c<<endl;
        Hanoi(n-1,b,a,c);//把n-1个圆盘从B柱借助A柱移动到C柱
    }
}
int main()
{
    cout<<"请输入盘子的个数: ";
    int n;
    cin>>n;
    Hanoi(n,'A','B','C');
    return 0;
}

合并排序

#include <iostream>
#include <math.h>
using namespace std;
int a[100];
int merges[100];//保存排序后的序列
//合并两个有序子序列
void Merge(int merges[],int a[],int left,int right,int middle)
{
    int i=left;
    int j=middle+1;
    int k=left;
    while(i<=middle&&j<=right)
    {
        if(a[i]<=a[j])
        {
            merges[k++]=a[i++];
        }
        else
        {
            merges[k++]=a[j++];
        }
    }
    //当有一方没有元素时，继续保存另一方剩下的元素
    while(i<=middle)merges[k++]=a[i++];
    while(j<=right)merges[k++]=a[j++];
    //更新排序后的a数组
    for(int i=left;i<=right;i++)
    {
        a[i]=merges[i];
    }
}
void MergeSort(int a[],int left,int right)
{
    if(left<right)
    {
        int middle=(left+right)>>1;
        MergeSort(a,left,middle);
        MergeSort(a,middle+1,right);
        Merge(merges,a,left,right,middle);
    }
}
int main()
{
    cout<<"请输入数组元素个数: ";
    int n;
    cin>>n;
    cout<<"请输入要排序的数组: ";
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    MergeSort(a,0,n-1);
    cout<<"排序后的数组是: ";
    for(int i=0;i<n;i++)
    {
        cout<<a[i]<<" ";
    }
    return 0;
}
