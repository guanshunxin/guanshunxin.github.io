最小延迟调度

【问题描述】
给定等待服务的客户集合A={1,2，… , n}, 预计对客户i的服务时间是ti, 该客户希望的完成时间是di, 即T=<t1,t2，…,tn>, D=<d1, d2,… ,dn>. 如果对客户i的服务在di之前结束, 那么对客户i的服务没有延迟;如果在di之后结束, 那么这个服务就被延迟了, 延迟的时间等于该服务结束时间减去di . 假设ti, di都是正整数, 一个调度函数f : AN，f(i)为对客户i的服务开始的时间, 要求所有区间(f(i), f(i)+ti)互不重叠. 一个调度f 的最大延迟是所有客户延迟时间的最大值. 求最大延迟达到最小的调度f .
【输入形式】客户数n，各客户服务时间及期望完成时间
【输出形式】最小延迟时间调度安排及延迟时间
【样例输入】
5
5 8 4 10 3
10 12 15 11 20
【样例输出】
1 4 2 3 5
12

#include <iostream>

using namespace std;
void delay(int n,int t[],int d[])
{
    int id[10];
    for(int i=1;i<=n;i++)
    {
        id[i]=i;
    }
    for(int i=1;i<n;i++)
    {
        for(int j=1;j<n-i+1;j++)
        {
            if(d[j]>d[j+1])
            {
                int temp=t[j];
                int temp2=d[j];
                int temp3=id[j];
                t[j]=t[j+1];
                d[j]=d[j+1];
                id[j]=id[j+1];
                t[j+1]=temp;
                d[j+1]=temp2;
                id[j+1]=temp3;
            }
        }
    }
    int delayAll=0;
    int Max=0;
    for(int i=1;i<=n;i++)
    {
        delayAll+=t[i];
        if((delayAll-d[i])>Max)
        {
            Max=delayAll-d[i];
        }
    }
    cout<<"延迟时间是: "<<Max<<endl;
    cout<<"事件服务顺序是: ";
    for(int i=1;i<=n;i++)
    {
        cout<<id[i]<<" ";
    }

}
int main()
{
    cout<<"请输入事件总数: ";
    int n;
    cin>>n;
    int t[20],d[20];
    cout<<"请输入事件服务时间: ";
    for(int i=1;i<=n;i++)
    {
        cin>>t[i];
    }
    cout<<"请输入事件截止时间: ";
    for(int i=1;i<=n;i++)
    {
        cin>>d[i];
    }
    delay(n,t,d);

    return 0;
}

找零钱问题

#include<iostream>
using namespace std;
//取它们的最小值，谁有解取谁，如果都没有解，那么都是-1 ，返回谁都可以
int min(int num1,int num2)
{
    //都有解，选小的
    if(num1!=-1 && num2!=-1)
    {
        if(num1<num2)
            return num1;
        else
            return num2;
    }
    //有一个没有解，或都没有解
    else
    {
        if(num2==-1)
            return num1;
        return num2;
    }
}
int main()
{
    int n;//用来接收需要支付的金额
    int M;//零钱种类数
    cout<<"请输入零钱种类数: ";
    cin>>M;
    cout<<"请输入需要支付的金额：";
    cin>>n;
    int money[M+1];//零钱种类
    int a[M+1][n+1];//表格
    int count[M+1]={0};//每种零钱数量
    int i,j;
    cout<<"请输入零钱种类: ";
    for(int i=1;i<=M;i++)
    {
        cin>>money[i];
    }
    //初始化表格
    for(i=0;i<=M;i++)
        a[i][0]=0;
    for(j=1;j<=n;j++)
        a[0][j]=-1;//表示无解

    //1、开始求解，填写表格
    for(i=1;i<=M;i++)   //从下标都为1  开始
    {
        for(j=1;j<=n;j++)
        {
            //1、判断支付金额是否大于等于当前硬币的面值，如果大于等于那么可以支付，否则为子问题的解
            //2、如果可以支付，判断子问题的解  和  使用该面值后的硬币数  谁少取谁
            if(j>=money[i])
                a[i][j]=min(a[i][j-money[i]]+1,a[i-1][j]);
            else
                a[i][j]=a[i-1][j];
        }
    }
    //2、求选择硬币的个数
    j=n;
    i=M;
    while(i>0)
    {
        if(a[i][j]!=a[i-1][j])
        {
            count[i]++;
            j-=money[i];
        }
        else i--;
    }
    //输出结果式
    cout<<n<<"=";
    bool sum=true;//第一个不需要加+
    for(int i=M;i>0;i--)
    {
        while(count[i]>0)
        {
            if(sum==true)
            {
                cout<<money[i];
                sum=false;
            }
            else
            {
                cout<<"+"<<money[i];
            }
            count[i]--;
        }
    }
    return 0;
}
