【问题描述】给定n个矩阵：A1, A2, …,An，其中Ai与Ai+1是可乘的，i=1，2…，n-1。确定计算矩阵连乘积的计算次序，使得依此次序计算矩阵连乘积需要的数乘次数最少。
【输入形式】输入数据为矩阵个数和每个矩阵规模
【输出形式】输出结果为计算矩阵连乘积的最少数乘次数和计算次序。
【样例输入】
5
30 35 15 5 10 20
样例输出】
11875
Multiply A2,2 and A3,3
Multiply A1,1 and A2,3
Multiply A4,4 and A5,5
Multiply A1,3 and A4,5

矩阵连乘

#include <iostream>
#define MaxSize 6
using namespace std;
int n;
int m[MaxSize][MaxSize];
int s[MaxSize][MaxSize];
int a[MaxSize];
void printmatrix(int left,int right)
{
    if(right-left==0)
    {
        return;
    }
    else
    {
        printmatrix(left,s[left][right]);
        printmatrix(s[left][right]+1,right);
        cout<<"Multiply A"<<left<<","<<s[left][right]<<"and A"<<s[left][right]+1<<","<<right<<endl;
    }

}
void matrix()
{
    for(int r=2;r<=n;r++)//矩阵长度
    {
        for(int i=1;i<=n-r+1;i++)//行循环
        {
            int j=i+r-1;//从第i个矩阵Ai开始,长度为r,则矩阵段为(Ai~Aj)
            //例如对(A2~A4),则i=2,j=4,下面一行的m[2][4]=m[3][4]+p[1]*p[2]*p[4],即A2(A3A4)
            m[i][j] = m[i+1][j]+a[i-1]*a[i]*a[j];
            s[i][j]=i;//记录断开点的索引
            for(int k=i+1;k<j;k++)
            {
                int t=m[i][k]+m[k+1][j]+a[i-1]*a[k]*a[j];
                //将矩阵段(Ai~Aj)分成左右2部分(左m[i][k],右m[k+1][j]),
                //再加上左右2部分最后相乘的次数(p[i-1] *p[k]*p[j])
                if(t<m[i][j])
                {
                    m[i][j]=t;
                    s[i][j]=k;
                }
            }
        }
    }
    for(int i=1;i<6;i++)
    {
        for(int j=1;j<6;j++)
        {
            cout<<m[i][j]<<" ";
        }
       cout<<endl;
    }
   cout<<"矩阵连乘最小次数："<<m[1][n]<<endl;
   cout<<"索引矩阵是："<<endl;
   for(int i=1;i<6;i++)
    {
        for(int j=1;j<6;j++)
        {
            cout<<s[i][j]<<" ";
        }
        cout<<endl;
    }
   cout<<"矩阵连乘的形式："<<endl;
   printmatrix(1,n);
}
int main()
{
    cout<<"请输入矩阵个数：";
    cin>>n;
    cout<<"请输入每个矩阵规模：";
    for(int i=0;i<=n;i++)
    {
        cin>>a[i];
    }
    matrix();
    return 0;
}

01背包问题

【问题描述】给定 n 个不同物品和一个背包，物品 i 的重量是 wi，价值为 vi，背包容量为 C，物品 i 只有两种选择：装入背包或不装入背包（0/1背包问题，0/1 knapsack problem）。如何选择装入背包的物品，使得装入背包中物品的总价值最大？
【输入形式】依次输入n，C，物品重量数组w，物品价值数组v
【输出形式】输出最大价值及选择的物品xi
【样例输入】
5
10
2 2 6 5 4
6 3 5 4 6
【样例输出】
15
x1 x2 x5

#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int n,c,w[20],v[20],V[20][20],x[20];
    cout<<"请输入物品种类: ";
    cin>>n;
    cout<<"请输入背包容量: ";
    cin>>c;
    for(int i=0;i<=n;i++)
    {
        V[i][0]=0;//背包容量为0该价值列为0
    }
    for(int j=0;j<=c;j++)
    {
        V[0][j]=0;//物品行为0该行价值为0
    }
    cout<<"请输入物品重量: ";
    for(int i=1;i<=n;i++)
    {
        cin>>w[i];
    }
    cout<<"请输入物品价值: ";
    for(int i=1;i<=n;i++)
    {
        cin>>v[i];
    }
    for(int i=1;i<=n;i++)//物品行
    {
        for(int j=1;j<=c;j++)//容量列
        {
            if(j<w[i])
            {
                V[i][j]=V[i-1][j];
                //当容量不足以装入i物品就等于前一个容量所装物品的价值
            }
            else
            {
                V[i][j]=max(V[i-1][j],V[i-1][j-w[i]]+v[i]);
                //不装入该物品和装入该物品价值作比较
                //要为该物品腾出w[i]的容量，
                //剩余容量为j-w[i]，就是在物品可装入为前i-1个物品且背包容量为j-w[i]时，
                //背包所装的价值加上新装入背包的物品价值就是总价值了
            }
        }
    }
    for(int j=c,i=n;i>0;i--)
    {
        if(V[i][j]>V[i-1][j])
        {
            x[i]=1;//在该容量下物品价值增加说明该物品装入背包
            j=j-w[i];//更新剩余背包容量
        }
        else
        {
            x[i]=0;
        }
    }
    cout<<"最大价值是: "<<V[n][c]<<endl;
    cout<<"放入的物品是: ";
    for(int i=1;i<=n;i++)
    {
        if(x[i]==1)
        {
            cout<<"x"<<i<<" ";
        }
    }
    cout<<endl;
    for(int i=0;i<=n;i++)
    {
        for(int j=0;j<=c;j++)
        {
            cout<<setw(2)<<V[i][j]<<" ";
        }
        cout<<endl;
    }
    return 0;
}

最长公共子序列

【问题描述】
给定一个序列，如果其中有些元素(也可能没有)被省略，则我们可以得到该序列的一个子序列。给定一个序列X = < > ，另一个序列Z满足条件，存在严格递增索引序列I=< >,使得对于所有j = 1,2,…,k, ，则称Z= < >是X的子序列。例如，Z = <a, b, f, c>是X = <a, b, c, f, b, c>的子序列。
给定两个序列X和Y，问题是求出X和Y的最大长度公共子序列的长度。
【输入形式】两个给定字符串
【输出形式】两个序列最大长度公共子序列的长度。
【样例输入】
abcfbc abfcab
【样例输出】
4

#include <iostream>

using namespace std;
char ss[100];
int sum=0;
void StructureSequence(int s[][100],char x[],int i,int j)
{
    if(i==0||j==0)
    {
        return;
    }
    if(s[i][j]==1)
    {
        ss[sum++]=x[i];
        StructureSequence(s,x,i-1,j-1);
    }
    else if(s[i][j]==2)
    {
        StructureSequence(s,x,i,j-1);
    }
    else
    {
        StructureSequence(s,x,i-1,j);
    }

}
void LCS(int m,int n,char x[],char y[],int c[][100],int s[][100])
{
    //定义边界为0
    //s中设置标记1,2,3表示三种情况
    for(int i=0;i<=m;i++)
    {
        c[i][0]=0;
    }
    for(int j=0;j<=n;j++)
    {
        c[0][j]=0;
    }
    for(int i=1;i<=m;i++)
    {
        for(int j=1;j<=n;j++)
        {
            if(x[i]==y[j])
            {
                //表示序列X和序列Y的最后项x[i],y[j]已被选入最长公共子序列
                c[i][j]=c[i-1][j-1]+1;
                s[i][j]=1;
            }
            //当序列Y少了最后一项或序列X少最后一项最长序列长度做比较
            //最长的就为此时的最长子序列
            //因为最后一项不等所以选出其中一个序列少最后一项的最长公共子序列即为此时的最长公共子序列
            else if(c[i][j-1]>c[i-1][j])
            {
                //表示最长公共子序列是Xi,Yj-1的最长公共子序列
                c[i][j]=c[i][j-1];
                s[i][j]=2;
            }
            else
            {
                //表示最长公共子序列是Xi-1,Yj的最长公共子序列
                c[i][j]=c[i-1][j];
                s[i][j]=3;
            }
        }
    }
    StructureSequence(s,x,m,n);
}
int main()
{
    int m,n;
    char x[20],y[20];
    int s[100][100],c[100][100];
    cout<<"请输入序列X的长度: ";
    cin>>m;
    cout<<"请输入序列X: ";
    for(int i=1;i<=m;i++)
    {
        cin>>x[i];
    }
    cout<<"请输入序列Y的长度: ";
    cin>>n;
    cout<<"请输入序列Y: ";
    for(int i=1;i<=n;i++)
    {
        cin>>y[i];
    }
    LCS(m,n,x,y,c,s);
    cout<<"最长公共子序列长度是: "<<sum<<endl;
    cout<<"最长公共子序列: ";
    for(int i=sum-1;i>=0;i--)
    {
        cout<<ss[i]<<" ";
    }
    return 0;
}

最优二叉搜索树

【问题描述】
数据集 S＝< 1, 2, …, n >
存取概率分布P = < a0, b1, a1, b2, … , ai, bi+1, …, bn, an >(b0 = 0), 求一棵最优的(即平均比较次数最少的)二分检索树.
假设 m[i,j] 是相对于输入 S[i,j] 和 P[i,j] 的最优二叉搜索树的平均比较次数，令m[i,i-1]=0.
【输入形式】输入数据结点个数n，空袭结点及数据结点存取概率
【输出形式】最优二叉搜索树的中序遍历结果
【样例输入】
5
0.04 0.1 0.02 0.3 0.02 0.1 0.05 0.2 0.06 0.1 0.01
【样例输出】
2 1 4 3 5

#include <iostream>
#include <iomanip>
using namespace std;

const int MaxVal = 9999;
int n;
//搜索到根节点和虚拟键的概率
double p[20];
double q[20];
int root[20][20];//记录根节点
double w[20][20];//子树概率总和
double e[20][20];//子树期望代价
void optimalBST(double *p,double *q,int n)
{
    //初始化只包括虚拟键的子树
    for (int i = 1; i <= n + 1; ++i)
    {
        w[i][i - 1] = q[i - 1];
        e[i][i - 1] = q[i - 1];
    }

    //由下到上，由左到右逐步计算
    for (int len = 1; len <= n; len++)
    {
        for (int i = 1; i <= n - len + 1; i++)
        {
            int j = i + len - 1;
            e[i][j] = MaxVal;
            w[i][j] = w[i][j - 1] + p[j] + q[j];
            //求取最小代价的子树的根
            for (int k = i; k <= j; k++)
            {
                double temp = e[i][k - 1] + e[k + 1][j] + w[i][j];
                if (temp < e[i][j])
                {
                    e[i][j] = temp;
                    root[i][j] = k;
                }
            }
        }
    }
}

//打印最优二叉查找树的结构
//打印出[i,j]子树，它是根r的左子树和右子树
void printOptimalBST(int i,int j,int r)
{
    int rootChild = root[i][j];//子树根节点
    if (rootChild == root[1][n])
    {
        //输出整棵树的根
        cout << rootChild<<" ";
        printOptimalBST(i,rootChild - 1,rootChild);
        printOptimalBST(rootChild + 1,j,rootChild);
        return;
    }

    if (j <=i - 1)
    {
        return;
    }
    else//遇到内部结点
    {
        cout<<rootChild<<" ";
    }

    printOptimalBST(i,rootChild - 1,rootChild);
    printOptimalBST(rootChild + 1,j,rootChild);
}
void print1()
{
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=n;j++)
        {
            cout<<setw(4)<<e[i][j]<<" ";
        }
        cout<<endl;
    }
    cout<<endl;
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=n;j++)
        {
            cout<<setw(4)<<w[i][j]<<" ";
        }
        cout<<endl;
    }
}
int main()
{
    cout<<"请输入数据节点个数: ";
    cin>>n;
    double a;
    int p1=1;
    int q1=0;
    for(int i=0;i<=n*2;i++)
    {
        cin>>a;
        if(i%2==0)
        {
            q[q1++]=a;
        }
        else
        {
            p[p1++]=a;
        }
    }
    optimalBST(p,q,n);
    cout << "最优二叉树结构：" << endl;
    print1();
    printOptimalBST(1,n,-1);
}
