#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
void myPrint(int a)
{
    cout<<a<<endl;
}
void test()
{
    vector<int> v;
    v.push_back(10);
    v.push_back(20);
    v.push_back(30);
    v.push_back(40);
    //起始迭代器，指向容器中的第一个元素
    vector<int>::iterator itBegin=v.begin();
    //结束迭代器，指向容器中最后一个元素后一个位置
    vector<int>::iterator itEnd=v.end();
    //第一种遍历方式
    while(itBegin!=itEnd)
    {
        cout<<*itBegin<<endl;
        itBegin++;
    }
    //第二种遍历方式
    for(vector<int>::iterator it=v.begin();it!=v.end();it++)
    {
        cout<<*it<<endl;
    }
    //第三种遍历方式
    for_each(v.begin(),v.end(),myPrint);
}
int main()
{
    test();
    return 0;
}

互换容器

#include <iostream>
#include<vector>
using namespace std;
void printVetor(vector<int>&v)
{
    for(vector<int>::iterator it=v.begin();it!=v.end();it++)
    {
        cout<<*it<<" ";
    }
    cout<<endl;
}
void test()
{
    vector<int>v1;
    for(int i=0;i<10;i++)
    {
        v1.push_back(i);
    }
    cout<<"交换前："<<endl;
    printVetor(v1);
    vector<int>v2;
    for(int i=10;i>0;i--)
    {
        v2.push_back(i);
    }
    printVetor(v2);
    cout<<"交换后："<<endl;
    v1.swap(v2);
    printVetor(v1);
    printVetor(v2);
}
//2、实际用途
//巧用swap可以收缩空间
void test1()
{
    vector<int>v3;
    for(int i=0;i<100000;i++)
    {
        v3.push_back(i);
    }
    cout<<"v3的容量是："<<v3.capacity()<<endl;
    cout<<"v3的大小是："<<v3.size()<<endl;
    //重新指定大小
    v3.resize(3);
    cout<<"v3的容量是："<<v3.capacity()<<endl;
    cout<<"v3的大小是："<<v3.size()<<endl;
    //巧用swap收缩内存
    cout<<"巧用swap收缩内存"<<endl;
    vector<int>(v3).swap(v3);
    cout<<"v3的容量是："<<v3.capacity()<<endl;
    cout<<"v3的大小是："<<v3.size()<<endl;
}
int main()
{
    test();
    test1();
    return 0;
}

预留空间

#include <iostream>
#include<vector>
using namespace std;
void test()
{
    vector<int>v;
    int num=0;
    int *p=NULL;
    for(int i=0;i<100000;i++)
    {
        v.push_back(i);
        if(p!=&v[0])
        {
            //重新开辟时首地址会改变
            p=&v[0];
            num++;
        }
    }
    cout<<"动态扩展了："<<num<<" 次"<<endl;
}

        v1.push_back(i);
    }
    cout<<"交换前："<<endl<<"v：";
    for_each(v.begin(),v.end(),print01);
    cout<<endl<<"v1：";
    for_each(v1.begin(),v1.end(),print01);
    cout<<endl;
    swap(v,v1);
    cout<<"交换后："<<endl<<"v：";
    for_each(v.begin(),v.end(),print01);
    cout<<endl<<"v1：";
    for_each(v1.begin(),v1.end(),print01);
    return 0;
}
