线性表的顺序表示

#include <iostream>
#define MAXSIZE 100
using namespace std;
//定义储存结构
typedef struct
{
    int *elem;//元素的储存空间
    int length;//顺序表包含元素长度
} SqList; //结构类型，整个结构的名称

//初始化顺序表
void InitSqList(SqList &L)
{
    L.elem=new int[MAXSIZE];//分配储存空间
    L.length=0;//初始化长度为0
}

//线性表长度
void ListLength(SqList L)
{
    cout<<"线性表长度为: "<<L.length<<endl;
}
//判断是否为空
void isEmpty(SqList L)
{
    if(L.length==0)
    {
        cout<<"线性表为空！"<<endl;
    }
    else
    {
        cout<<"线性表不为空"<<endl;
    }
}

//清空线性表
void clearList(SqList &L)
{
    L.length=0;
}
//插入
void ListInsert(SqList &L,int i,int e)
{
    //判断是否越界i>=1&&i<=length+1
    if(i>L.length+1||i<1)
    {
        cout<<"i值不合法请重新输入!"<<endl;
    }
    //判断储存空间是否已满
    else if(L.length==MAXSIZE)
    {
        cout<<"储存空间已满！"<<endl;
    }
    else
    {
        //该位置后面元素后移1位
        for(int j=L.length; j>=i; j--)
        {
            L.elem[j]=L.elem[j-1];
        }
        L.elem[i-1]=e;//修改该位置元素
        L.length++;//长度加1
        cout<<"插入成功！"<<endl;
    }
}

//查找
void LocateItem(SqList L,int e)
{
    for(int i=0;i<L.length;i++)
    {
        if(L.elem[i]==e)
        {
            cout<<"找到该元素 "<<e<<" 的位置为: "<<i+1<<endl;
        }
    }
}

//删除
void DeleteList(SqList &L,int i)
{
    //判断是否越界i>=1&&i<=length+1
    if(i>L.length+1||i<1)
    {
        cout<<"i值不合法请重新输入!"<<endl;
    }
    else
    {
        for(int j=i-1; j<L.length-1; j++)
        {
            L.elem[j]=L.elem[j+1];
        }
        L.length--;//长度减1
        cout<<"删除成功！"<<endl;
    }
}

//显示顺序表元素
void ShowList(SqList L)
{
    for(int i=0;i<L.length;i++)
    {
        cout<<L.elem[i]<<" ";
    }
    cout<<endl;
}

//销毁线性表
void DestoryList(SqList &L)
{
    if(L.elem) delete[] L.elem;
    L.length=0;
    L.elem=NULL;
    cout<<"摧毁成功！"<<endl;
}
//菜单
void menu()
{
    cout<<"####### 1 插入       #######"<<endl;
    cout<<"####### 2 清空       #######"<<endl;
    cout<<"####### 3 是否为空   #######"<<endl;
    cout<<"####### 4 线性表长度 #######"<<endl;
    cout<<"####### 5 查找       #######"<<endl;
    cout<<"####### 6 删除       #######"<<endl;
    cout<<"####### 7 摧毁       #######"<<endl;
    cout<<"####### 8 显示       #######"<<endl;
    cout<<"####### 9 退出       #######"<<endl;
}
int main()
{
    SqList L;//创建结构体变量
    InitSqList(L);//初始化
    int i=0,e=0;
    menu();
    char select;
    while(true)
    {
        cout<<"请输入选择：";
        cin>>select;
        if(select=='1')
        {
            cout<<"请输入位置和元素：";
            cin>>i>>e;
            ListInsert(L,i,e);
        }
        if(select=='2')
        {
            clearList(L);
        }
        if(select=='3')
        {
            isEmpty(L);
        }
        if(select=='4')
        {
            ListLength(L);
        }
        if(select=='5')
        {
            cout<<"请输入要查找的元素：";
            cin>>e;
            LocateItem(L,e);
        }
        if(select=='6')
        {
            cout<<"请输入要删除元素的位置：";
            cin>>i;
            DeleteList(L,i);
        }
        if(select=='7')
        {
            DestoryList(L);
        }
        if(select=='8')
        {
            ShowList(L);
        }
        if(select=='9')
        {
            cout<<"退出！"<<endl;
            break;
        }
    }
    return 0;
}

线性表的链式表示

#include <iostream>

using namespace std;
typedef struct LNode
{
    int data;
    struct LNode *next;
} LNode,*LinkList;
//typedef struct LNode LNode; 将结构体重命名为LNode
//typedef struct LNode *LinkList; 将结构体指针重命名为LinkList
//这里对LinkList的理解参考
//https://zhuanlan.zhihu.com/p/421431276
//https://zhuanlan.zhihu.com/p/410715721
//当参数为LinkList &L时，意味着需要改变或操作L这个指针本身
//相当是实参修改L的指向 指针的引用当参数是
//既可以改变指针指向的内容，又可以修改指针本身。
//初始化
void InitList(LinkList &L)
{
    L=new LNode;//生成新结点作为头结点，用头指针L指向头结点
    L->next=NULL;//头结点的指针域置空
}
//取值
void GetElem(LinkList L,int i,int &e)
{
    LinkList p;//创建指针
    p=L->next;//指向首元结点
    int j=1;
    while(p&&j<i)//直到当p为空或指向第i个元素
    {
        p=p->next;//指向下一个节点
        j++;
    }
    if(!p||j>i)//分别是i大于表长和当开始输入的i<j时，就是i<1
    {
        cout<<"i值不合法请重新输入！"<<endl;
    }
    else
    {
        e=p->data;
        cout<<"取值成功！为："<<e<<endl;
    }
}
//按值查找
LNode *LocateElem(LinkList L,int e)
{
    LinkList p;
    p=L->next;
    while(p&&p->data!=e)
    {
        p=p->next;
    }
    return p;//返回指针p
}
//插入
void ListInsert(LinkList &L,int i,int e)
{
    LinkList p;
    p=L;//指向头节点以防插入到首元节点
    int j=0;
    while(p&&(j<i-1))
    {
        p=p->next;
        j++;
    }
    if(!p||j>i-1)
    {
        cout<<"i值不合法请重新输入！"<<endl;
    }
    else
    {
        LinkList s;
        s=new LNode;//生成新节点以便插入
        s->data=e;
        s->next=p->next;
        p->next=s;
        cout<<"插入成功！"<<endl;
    }
}
//删除
void ListDelete(LinkList &L,int i)
{
    LinkList p;
    p=L;
    int j=0;
    //p->next防止p为空节点
    //若为p当p为空节点下面的判断条件p->next引用空指针引起删除失败
    while((p->next)&&(j<i-1))
    {
        p=p->next;
        j++;
    }
    if(!(p->next)||(j>i-1))
    {
        cout<<"删除位置不合理！"<<endl;
    }
    else
    {
        LinkList q=p->next;
        p->next=q->next;
        delete q;
    }
}
//前插法创建单链表
void CreateList(LinkList &L)
{
    int n;
    cout<<"请输入单链表长度：";
    cin>>n;
    cout<<"请输入元素：";
    for(int i=0; i<n; i++)
    {
        LinkList p=new LNode;
        cin>>p->data;
        p->next=L->next;
        L->next=p;
    }
}
//后插法创建单链表
void CreateList_R(LinkList &L)
{
    int n;
    cout<<"请输入单链表长度：";
    cin>>n;
    LinkList r;
    r=L;
    cout<<"请输入元素：";
    for(int i=0; i<n; i++)
    {
        LinkList p=new LNode;
        cin>>p->data;
        r->next=p;
        p->next=NULL;
        r=p;
    }
}
//清空线性表
void clearList(LinkList &L)
{
    LinkList p,q;
    p=L->next;
    while(p)
    {
        q=p->next;
        delete p;
        p=q;
    }
    L->next=NULL;
}
//求线性表的长度
void ListLength(LinkList L)
{
    LinkList p=L->next;
    int Count=0;
    while(p)
    {
        Count++;
        p=p->next;
    }
    cout<<"线性表的长度是："<<Count<<endl;
}

//摧毁线性表
void DestroyList(LinkList &L)
{
    LinkList p;
    while(L)
    {
        p=L;
        L=L->next;
        delete p;
    }
    cout<<"线性表摧毁成功！"<<endl;
}
void showList(LinkList L)
{
    LinkList p=L->next;
    while(p)
    {
        cout<<p->data<<" ";
        p=p->next;
    }
    cout<<endl;
}
void menu()
{
    cout<<"####### 1 取值       #######"<<endl;
    cout<<"####### 2 按值查找   #######"<<endl;
    cout<<"####### 3 插入       #######"<<endl;
    cout<<"####### 4 删除       #######"<<endl;
    cout<<"####### 5 后插       #######"<<endl;
    cout<<"####### 6 前插       #######"<<endl;
    cout<<"####### 7 清空       #######"<<endl;
    cout<<"####### 8 长度       #######"<<endl;
    cout<<"####### 0 摧毁       #######"<<endl;
    cout<<"####### S 显示       #######"<<endl;
    cout<<"####### T 退出       #######"<<endl;
}
int main()
{
    LinkList L;
    int i,e;
    InitList(L);
    char select;
    menu();
    while(true)
    {
        cout<<"请输入选项：";
        cin>>select;
        if(select=='1')
        {
            cout<<"请输入取值位置和元素：";
            cin>>i>>e;
            GetElem(L,i,e);
        }
        if(select=='2')
        {
            cout<<"请输入要查找的元素：";
            cin>>e;
            LinkList p=LocateElem(L,e);
            if(p)
            {
                cout<<"查找到的值为："<<p->data<<endl;
            }
            else
            {
                cout<<"查找失败！"<<endl;
            }
        }
        if(select=='3')
        {
            cout<<"请输入插入位置和元素：";
            cin>>i>>e;
            ListInsert(L,i,e);
        }
        if(select=='4')
        {
            cout<<"请输入需要删除的位置：";
            cin>>i;
            ListDelete(L,i);
        }
        if(select=='5')
        {
            CreateList_R(L);
        }
        if(select=='6')
        {
            CreateList(L);
        }
        if(select=='7')
        {
            clearList(L);
        }
        if(select=='8')
        {
            ListLength(L);
        }
        if(select=='0')
        {
            DestroyList(L);
        }
        if(select=='S')
        {
            showList(L);
        }
        if(select=='T')
        {
            break;
        }
    }
    return 0;
}

交集、并集、差集

#include <iostream>

using namespace std;
typedef struct LNode
{
    int data;
    struct LNode *next;
} LNode,*LinkList;
void InitList(LinkList &L)
{
    L=new LNode;//生成新结点作为头结点，用头指针L指向头结点
    L->next=NULL;//头结点的指针域置空
}
void CreateLNode(LinkList &L)
{
    cout<<"请输入创建线性表的长度: ";
    int n;
    cin>>n;
    LinkList r;
    r=L;
    cout<<"请输入线性表内的元素: ";
    for(int i=0; i<n; i++)
    {
        LinkList p=new LNode;
        cin>>p->data;
        r->next=p;
        p->next=NULL;
        r=p;
    }
}
//交集
void Mix(LinkList &La,LinkList &Lb,LinkList &Lc)
{
    LinkList pa=La->next;//pa和pb作为工作指针
    LinkList pb=Lb->next;
    LinkList pc;
    Lc=La;//将La的头结点作为Lc的头结点
    pc=La;
    while(pa&&pb)
    {
        if(pa->data==pb->data)
        {
            pc->next=pa;
            pc=pa;
            pa=pa->next;
            LinkList u=pb;
            pb=pb->next;
            delete u;
        }
        else if(pa->data<pb->data)
        {
            LinkList u=pa;
            pa=pa->next;
            delete u;
        }
        else
        {
            LinkList u=pb;
            pb=pb->next;
            delete u;
        }
    }
    while(pa)
    {
        LinkList u=pa;
        pa=pa->next;
        delete u;
    }
    while(pb)
    {
        LinkList u=pb;
        pb=pb->next;
        delete u;
    }
    pc->next=NULL;
    delete Lb;
}
//并集
void Union(LinkList &La,LinkList &Lb,LinkList &Lc)
{
    LinkList pa=La->next;//pa和pb作为工作指针
    LinkList pb=Lb->next;
    LinkList pc;
    Lc=La;//将La的头结点作为Lc的头结点
    pc=La;
    while(pa||pb)
    {
        if(pa&&pb)
        {
            if(pa->data==pb->data)
            {
                pc->next=pa;
                pc=pa;
                pa=pa->next;
                LinkList u=pb;
                pb=pb->next;
                delete u;
            }
            else if(pa->data<pb->data)
            {
                pc->next=pa;
                pc=pa;
                pa=pa->next;
            }
            else
            {
                pc->next=pb;
                pc=pb;
                pb=pb->next;
            }
        }
        else if(pa)
        {
            pc->next=pa;
            pc=pa;
            pa=pa->next;
        }
        else
        {
            pc->next=pb;
            pc=pb;
            pb=pb->next;
        }
    }
    while(pa)
    {
        LinkList u=pa;
        pa=pa->next;
        delete u;
    }
    while(pb)
    {
        LinkList u=pb;
        pb=pb->next;
        delete u;
    }
    pc->next=NULL;
    delete Lb;
}
//差集
void Difference_set(LinkList &La,LinkList &Lb,LinkList &Lc)
{
    LinkList pa=La->next;//pa和pb作为工作指针
    LinkList pb=Lb->next;
    LinkList pc;
    Lc=La;//将La的头结点作为Lc的头结点
    pc=La;
    while(pa)
    {
        if(pa&&pb)
        {
            if(pa->data==pb->data)
            {
                LinkList u=pa;
                pa=pa->next;
                delete u;
                u=pb;
                pb=pb->next;
                delete u;
            }
            else if(pa->data<pb->data)
            {
                pc->next=pa;
                pc=pa;
                pa=pa->next;
            }
            else
            {
                pc->next=pb;
                pc=pb;
                LinkList u=pb;
                pb=pb->next;
                delete u;
            }
        }
        else
        {
            pc->next=pa;
            pc=pa;
            pa=pa->next;
        }
    }
    while(pa)
    {
        LinkList u=pa;
        pa=pa->next;
        delete u;
    }
    while(pb)
    {
        LinkList u=pb;
        pb=pb->next;
        delete u;
    }
    pc->next=NULL;
    delete Lb;
}
int main()
{
    cout<<"****1 求交集 ****"<<endl;
    cout<<"****2 求并集 ****"<<endl;
    cout<<"****3 求差集 ****"<<endl;
    cout<<"请输入你的选择: ";
    int a;
    cin>>a;
    LinkList La,Lb,Lc;
    InitList(La);
    InitList(Lb);
    cout<<"创建顺序表La: "<<endl;
    CreateLNode(La);
    cout<<"创建顺序表Lb: "<<endl;
    CreateLNode(Lb);
    if(a==1)
    {
        Mix(La,Lb,Lc);
        cout<<"两个线性表的交集是: ";
    }
    else if(a==2)
    {
        Union(La,Lb,Lc);
        cout<<"两个线性表的并集是: ";
    }
    else if(a==3)
    {
        Difference_set(La,Lb,Lc);
        cout<<"两个线性表的差集是: ";
    }
    else
    {
        cout<<"输入错误！"<<endl;
        return 0;
    }
    LinkList pc;
    pc=Lc->next;
    while(pc)
    {
        cout<<pc->data<<" ";
        pc=pc->next;
    }
    cout<<endl;
    return 0;
}
